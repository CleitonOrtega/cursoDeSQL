/*

	FILTRANDO DADOS COM O WHERE E LIKE

*/

WHERE SEXO = 'M';

SELECT NOME, ENDERECO FROM TB_CLNT
WHERE SEXO = 'F';

SELECT NOME, SEXO FROM TB_CLNT
WHERE ENDERECO = 'RJ';



/*

	UTILIZANDO O LIKE

SELECT NOME, SEXO FROM TB_CLNT
